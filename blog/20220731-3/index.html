ps：js文件可能需要引用在正文中，也就是<body>和</body>之间。

下面是带注释讲解的js代码


// JavaScript Document
document.write("<div id=\"divt\"></div>");
function startTime(){
	var today = new Date();
	var year = today.getFullYear();
	var month = today.getMonth() + 1;
	var day = today.getDate();
	var hh = today.getHours();
	var mm = today.getMinutes();
	var ss = today.getSeconds();
	month = checkTime(month);
	day = checkTime(day);
	hh = checkTime(hh);
	mm = checkTime(mm);
	ss = checkTime(ss);
	document.getElementById('divt').innerHTML = year + "." + month + "." + day + " " + hh + ":" + mm + ":" + ss;
	//这里分别为年月日时分秒，可以自行更改样式和精确度
	setTimeout('startTime()',100);
	//这里的100设置多久同步一次时间，单位毫秒，越小越精准但越吃资源（其实也没多少），不建议大于1000
}
function checkTime(i){
	if(i<10){
		i = "0" + i;
	}
	return i;
}
//这里是在个位数的前面补零
